  <div id="counter">Swear count: 0</div>

  <script>
    const counterEl = document.getElementById("counter");
    let count = 0;

    // üîä Lista Twoich plik√≥w z GitHuba
    const sounds = Array.from({ length: 16 }, (_, i) =>
      `https://mariadabrowskam.github.io/swearjar/${i + 1}.MP3`
    );

    // üß† Lista przekle≈Ñstw do wykrywania
    const swears = ["fuck", "shit", "bitch", "asshole", "bastard", "dick", "crap", "damn", "bollocks"];

    // üîä Funkcja do odtwarzania losowego d≈∫wiƒôku
    function playRandomSound() {
      const randomSound = sounds[Math.floor(Math.random() * sounds.length)];
      const audio = new Audio(randomSound);
      audio.volume = 1.0;
      audio.play().catch((e) => console.warn("Audio blocked:", e));
    }

    // üî• Animacja licznika
    function popCounter() {
      counterEl.classList.add("pop");
      setTimeout(() => counterEl.classList.remove("pop"), 400);
    }

    // üéôÔ∏è Start rozpoznawania mowy
    async function startRecognition() {
      try {
        const stream = await navigator.mediaDevices.getUserMedia({ audio: true });
        console.log("üé§ Mikrofon dzia≈Ça:", stream);

        const recognition = new (window.SpeechRecognition || window.webkitSpeechRecognition)();
        recognition.lang = "en-US";
        recognition.continuous = true;
        recognition.interimResults = false;

        recognition.onresult = (event) => {
          const transcript = event.results[event.results.length - 1][0].transcript.toLowerCase();
          console.log("üéôÔ∏è Heard:", transcript);

          if (swears.some((word) => transcript.includes(word))) {
            count++;
            counterEl.textContent = `Swear count: ${count}`;
            popCounter();
            playRandomSound();
            console.log("üß® Swear detected!");
          }
        };

        recognition.onerror = (err) => console.error("Recognition error:", err);
        recognition.onend = () => {
          console.log("üîÅ Restarting recognition...");
          recognition.start();
        };

        recognition.start();
      } catch (err) {
        console.error("‚ùå Brak dostƒôpu do mikrofonu:", err);
      }
    }

    startRecognition();
  </script>
